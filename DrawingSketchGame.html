Drawing Sketch Game

//Drawing Game void setup(){
size(800,800);
background(255);
strokeWeight(5);
println("Draw with the mouse!");
println("Default color is black.");
println("Use keys to change colors. Key q to return back to black.");
println("Alt to clear");
}

void draw(){
if(mousePressed||key=='q'){
stroke(0);
line(pmouseX,pmouseY,mouseX,mouseY);
}


if(keyCode==ALT)
background(255);
if(key=='r'){
stroke(255,0,0);
line(pmouseX,pmouseY,mouseX,mouseY);
}


else if(key=='o'){
stroke(255,165,0);
line(pmouseX,pmouseY,mouseX,mouseY);
}
else if(key=='y'){
stroke(255,255,0);
line(pmouseX,pmouseY,mouseX,mouseY);
}
else if(key=='g'){
stroke(0,255,0);
line(pmouseX,pmouseY,mouseX,mouseY);
}
else if(key=='b'){
stroke(0,0,255);
line(pmouseX,pmouseY,mouseX,mouseY);
}
else if(key=='v'){
stroke(128,0,128);
line(pmouseX,pmouseY,mouseX,mouseY);
}
else if(key=='p'){
stroke(165,42,42);
line(pmouseX,pmouseY,mouseX,mouseY);
}
else if(key=='w'){
stroke(120);
line(pmouseX,pmouseY,mouseX,mouseY);
}
}

  • key button "q": black
  • key button "r": red
  • key button "o": orange
  • key button "y": yellow
  • key button "g": green
  • key button "b": blue
  • key button "v": purple
  • key button "p": brown
  • key button "w": gray
  • key button "ALT": clears the board
